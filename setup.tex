% --- Sommaire stylisé : timeline verticale ---
% Une ligne (#1..#4) de la timeline : highlight | index | côté(1=droite,-1=gauche) | libellé
\newcommand{\tlrow}[4]{%
  \pgfmathsetmacro{\tly}{-(#2-1)*1.15}%
  \pgfmathsetmacro{\tlcx}{#3*1.0}%
  \pgfmathsetmacro{\tllx}{#3*1.12}%
  \ifnum#3>0\relax \def\tlanchor{west}\else \def\tlanchor{east}\fi
  \ifnum#1=#2\relax
     \def\tldotstyle{curdot}\def\tllabcol{beamer_color}%
  \else
     \def\tldotstyle{dot}%
     \ifnum#1=0\relax \def\tllabcol{black}\else \def\tllabcol{black!45}\fi
  \fi
  \draw (0,\tly) -- (\tlcx,\tly);
  \node[font=\bfseries\large, anchor=\tlanchor, text=\tllabcol] at (\tllx,\tly) {#4};
  \node[\tldotstyle] at (0,\tly) {};
}
% Sommaire complet ; #1 = numéro de section à mettre en avant (0 = aucune)
\newcommand{\sommairetimeline}[1]{%
\vfill
\begin{center}
\begin{tikzpicture}[
   line width=1pt,
   dot/.style={circle,fill=beamer_color,draw=black,line width=1pt,minimum size=11pt,inner sep=0pt},
   curdot/.style={circle,fill=beamer_color,draw=black,line width=1.3pt,minimum size=17pt,inner sep=0pt},
]
  \draw (0,0.5) -- (0,{-(6-1)*1.15-0.5});
  \tlrow{#1}{1}{ 1}{Contexte général}
  \tlrow{#1}{2}{-1}{Gestion des incidents}
  \tlrow{#1}{3}{ 1}{Conception}
  \tlrow{#1}{4}{-1}{Réalisation}
  \tlrow{#1}{5}{ 1}{Tests et résultats}
  \tlrow{#1}{6}{-1}{Conclusion \& perspectives}
\end{tikzpicture}
\end{center}
\vfill
}
