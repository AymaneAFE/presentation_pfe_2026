\section{Comment utiliser le format de diapositives Turix}

\subsection{Texte}

\begin{frame}{Mettre le texte en forme}

\textit{Ceci est du texte en italique.} \textbf{Ceci est du texte en gras.} \texttt{Ceci est du texte en police machine à écrire.}

\tiny{Ceci est du texte minuscule.}
\scriptsize{Ceci est du texte très petit.}
\footnotesize{Ce texte a la taille d'une note de bas de page.}
\small{Ce texte est légèrement plus petit que la normale.}
\normalsize{Ce texte a une taille normale.}
\large{Ce texte est légèrement plus grand que la normale.}
\Large{Ce texte est grand.}
\LARGE{Ce texte est encore plus grand.}
\huge{Ce texte est énorme.}
\Huge{Ceci est la plus grande taille possible.}

\end{frame}

\subsection{Équations}

\begin{frame}{Insérer des équations}
    Exemple d'une équation numérotée :
    \begin{equation}
        \frac{\partial Q\left(z_k,t\right)}{\partial t}+gA_k\frac{\partial H\left(z_k,t\right)}{\partial z_k}+\mu_kQ\left(z_k,t\right)\left|Q\left(z_k,t\right)\right|+g\sin{a_k}=0,
        \label{eq:momentum}
    \end{equation}
    Une équation numérotée est référencée dans le texte via son \textit{étiquette} : l'Équation \eqref{eq:momentum} est connue sous le nom d'Équation de la Quantité de Mouvement.

    Exemple d'une équation non numérotée :
    $$\frac{\partial H\left(z_k,t\right)}{\partial t}+\frac{b_k^2}{gA_k}\frac{\partial Q\left(z_k,t\right)}{\partial z_k}=0.$$

    Voici un exemple d'équation dans le texte : $\mu_k=\frac{fQ_k}{2D_kA_k}$
\end{frame}
