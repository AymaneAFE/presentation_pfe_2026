\node[st, fill=beamer_color!28] (inv) at (0.5,-3)   {INVESTIGATING};
\node[st, fill=beamer_color!45] (wfr) at (6.5,-3) {WAITING\_FOR\_REVIEW};
\node[st, fill=beamer_color!62] (val) at (6.5,-5.0) {VALIDATED};
\node[term]                     (clo) at (0,-5.0)   {CLOSED};
% entrée / sortie
\node[circle, fill=beamer_color!70!black, minimum size=7pt, inner sep=0pt] (ini) at (-2.6,0) {};
\node[circle, draw=beamer_color!85!black, line width=1pt, minimum size=11pt, inner sep=0pt] (fin) at (-2.6,-5.0) {};

\draw[arr] (inv) -- (wfr) node[lbl,midway,above]{INVESTIGATION\_COMPLETE\\\itshape(agent RCA)};
\draw[arr] (wfr) -- (val) node[lbl,midway,right]{SRE\_VALIDATED\_\\HYPOTHESIS\\\itshape(opérateur)};

      node[lbl,midway,right]{INVESTIGATION\_\\COMPLETE\\\itshape(agent RCA)};
